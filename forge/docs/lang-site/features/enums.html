Enums stable

enum Name { Variants } — algebraic data types with match support

Algebraic data types with named variants and optional fields

Enums in Forge are algebraic data types that define a type with a fixed set of named variants. Each variant can optionally carry data of any type. This makes enums far more powerful than the simple integer enumerations found in C or Java. For example: `enum Shape { Circle(float), Rectangle(float, float), Point }`.

Enums are constructed by naming the variant: `let s = Shape.Circle(5.0)`. Pattern matching with `match` is the primary way to work with enums, and the compiler ensures every variant is handled. This exhaustiveness checking catches entire categories of bugs at compile time.

Enums are ideal for modeling state machines, command sets, message types, error categories, and any domain where a value is exactly one of several possibilities. The Result and Option types that power Forge's error handling and null safety are themselves enums under the hood.

If you are familiar with Rust enums, Swift enums with associated values, or Haskell data types, Forge enums work the same way. If you are coming from TypeScript, think of them as discriminated unions with compiler-enforced exhaustiveness.

Syntax

enum Name { Variant, Variant(fields) }

Examples

basic_enum.fg

/// # Basic Enum
///
/// Simple enum declaration and variant creation.
///

enum Shape {
  point
  circle(radius: float)
}

fn main() {
  let p = Shape.point
  let c = Shape.circle(radius: 3.14)
  match p {
    .point -> println("a point")
    .circle(r) -> println(`radius ${r}`)
  }
  match c {
    .point -> println("no radius")
    .circle(r) -> println("has radius")
  }
}

enum_chain_functions.fg

/// # Enum Chain Through Functions
///
/// Enum values passed through multiple function calls and matched repeatedly.
///

enum Level { low, medium, high, critical }

fn escalate(l: Level) -> Level {
  match l {
    .low -> Level.medium
    .medium -> Level.high
    .high -> Level.critical
    .critical -> Level.critical
  }
}

fn message(l: Level) -> string {
  match l {
    .low -> "ok"
    .medium -> "watch"
    .high -> "warning"
    .critical -> "ALERT"
  }
}

fn label(l: Level) -> string {
  match l {
    .low -> "low"
    .medium -> "medium"
    .high -> "high"
    .critical -> "critical"
  }
}

fn report(l: Level) -> string {
  let final_level = escalate(escalate(escalate(l)))
  `${label(final_level)}: ${message(final_level)}`
}

fn main() {
  println(report(Level.low))
  println(`${label(Level.low)}: ${message(Level.low)}`)
}

enum_in_function.fg

/// # Enum in Function
///
/// Passing enums to and returning enums from functions.
///

enum Priority {
  low
  medium
  high
  critical
}

fn upgrade(p: Priority) -> Priority {
  match p {
    .low -> Priority.medium
    .medium -> Priority.high
    .high -> Priority.critical
    .critical -> Priority.critical
  }
}

fn label(p: Priority) -> string {
  match p {
    .low -> "low"
    .medium -> "medium"
    .high -> "high"
    .critical -> "critical"
  }
}

fn main() {
  let p = Priority.low
  let p2 = upgrade(p)
  println(label(p2))
  let p3 = upgrade(upgrade(p2))
  println(label(p3))
}

enum_in_list.fg

/// # Enum in List
///
/// Storing enum values in a list and iterating with match.
///

enum Light { red, green, yellow }

fn action(l: Light) -> string {
  match l {
    .red -> "stop"
    .green -> "go"
    .yellow -> "caution"
  }
}

fn main() {
  let lights = [Light.red, Light.green, Light.yellow, Light.red]
  for l in lights {
    println(action(l))
  }
}

/// # Enum Match Exhaustive
///
/// Pattern matching on all enum variants.
///

enum Color {
  red
  green
  blue
}

fn name(c: Color) -> string {
  match c {
    .red -> "red"
    .green -> "green"
    .blue -> "blue"
  }
}

fn main() {
  println(name(Color.red))
  println(name(Color.green))
  println(name(Color.blue))
}

enum_mixed_types.fg

/// # Enum with Mixed Data Types
///
/// Enum variants carrying different-sized types (string vs int).
///

enum Tag {
  a(name: string)
  b(value: int)
}

fn describe(t: Tag) -> string {
  match t {
    .a(name) -> `got name: ${name}`
    .b(value) -> `got value: ${value}`
  }
}

fn main() {
  println(describe(Tag.a(name: "Alice")))
  println(describe(Tag.b(value: 42)))
}

enum_multi_string_fields.fg

/// # Enum with Multiple String Fields
///
/// Enum variant with two string fields, plus a variant with one string.
///

enum Name {
  full(first: string, last: string)
  nick(alias: string)
}

fn display(n: Name) -> string {
  match n {
    .full(f, l) -> `${f} ${l}`
    .nick(a) -> a
  }
}

fn main() {
  println(display(Name.full(first: "John", last: "Doe")))
  println(display(Name.nick(alias: "JD")))
}

enum_nested.fg

/// # Enum Nested Calls
///
/// Enum variants passed through multiple function calls.
///

enum Status {
  ok(value: int)
  fail(code: int)
}

fn describe(s: Status) -> string {
  match s {
    .ok(v) -> `ok: ${v}`
    .fail(c) -> `fail: ${c}`
  }
}

fn main() {
  let o = Status.ok(value: 42)
  let f = Status.fail(code: 0)
  println(describe(o))
  println(describe(f))
}

enum_recursive_match.fg

/// # Enum Recursive Function with Match
///
/// Enum matched recursively in a function that calls itself.
///

enum Expr {
  num(value: int)
  add(a: int, b: int)
}

fn eval(e: Expr) -> string {
  match e {
    .num(v) -> string(v)
    .add(a, b) -> `(${a} + ${b})`
  }
}

fn main() {
  let inner = eval(Expr.add(a: 1, b: 2))
  println(`(${inner} + 3)`)
}
