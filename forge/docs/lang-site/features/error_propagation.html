Error Propagation stable

Result types with ? propagation and catch blocks

Result types with Ok/Err constructors, ? propagation, and catch blocks

Error propagation in Forge uses the `?` operator to bubble errors up the call stack. When a function returns a Result type, appending `?` to the call either unwraps the success value or immediately returns the error to the caller. This eliminates the verbose `if err != nil` checks found in Go and the sprawling try/catch blocks of Java.

The `?` operator can only be used inside functions that themselves return a Result type. The compiler enforces this rule, so you always know whether a function can fail by looking at its return type. There are no hidden error paths.

For functions that need to handle errors rather than propagate them, the `catch` pattern provides structured error handling. This gives you the control of try/catch when you need it, while the `?` operator handles the common case of simply passing errors upward.

This design is directly inspired by Rust's `?` operator and Result type. It provides the same safety guarantees — every error must be explicitly handled or propagated — while keeping the syntax lightweight. Compared to exceptions, it makes error paths visible in type signatures and prevents the "exception from nowhere" problem.

Syntax

expr?
catch { body }

Examples

catch_different_types.fg

/// # Catch Different Types
///
/// Functions can return different error strings while sharing the same Result type.
///

fn read_file(path: string) -> Result<string, string> {
  if path == "valid.txt" { Ok("file ok") } else { Err("file not found") }
}

fn fetch_url(url: string) -> Result<string, string> {
  if url == "good" { Ok("data") } else { Err("network timeout") }
}

fn main() {
  let f = read_file("valid.txt") catch (e) { `caught: ${e}` }
  println(f)

  let n = fetch_url("bad") catch (e) { `caught: ${e}` }
  println(n)

  println("done")
}

catch_expression.fg

/// # Catch Expression
///
/// Use catch to handle Result errors inline.
///

fn greet(name: string) -> Result<string, string> {
  if name.length == 0 {
    Err("empty name")
  } else {
    Ok(`hello, ${name}!`)
  }
}

fn divide(a: float, b: float) -> Result<float, string> {
  if b == 0.0 { Err("division by zero") } else { Ok(a / b) }
}

fn main() {
  let msg = greet("world") catch (e) { `error: ${e}` }
  println(msg)

  let result = divide(1.0, 0.0) catch (e) {
    println(`error: ${e}`)
    0.0
  }
  println(string(result))
}

/// # Catch Without Binding
///
/// Catch blocks can omit the error binding when
/// the error value is not needed.
///

fn risky(x: int) -> Result<int, string> {
  if x > 0 { Ok(x) } else { Err("bad") }
}

fn main() {
  let a = risky(42) catch { 0 }

  let b = risky(-1) catch { -1 }

}

/// # Catch Re-Err
///
/// Catch handler can return a new Err via `return Err(...)`,
/// letting the outer catch handle it.
///

fn inner() -> Result<int, string> {
  Err("inner err")
}

fn outer() -> Result<int, string> {
  let v = inner() catch (e) { return Err("outer err") }
  Ok(v)
}

fn main() {
  let r = outer() catch (e) { 0 }
  println(`final: ${string(r)}`)
}

/// # Error Handling Pipeline
///
/// Combining Result types with pattern matching
/// for robust error handling chains.
///

fn try_parse(input: string) -> Result<int, string> {
  if input == "21" { Ok(21) } else { Err("not a number") }
}

fn parse_and_double(input: string) -> string {
  match try_parse(input) {
    Ok(n) -> "doubled: " + string(n * 2)
    Err(e) -> "error: " + e
  }
}

fn main() {
  println(parse_and_double("21"))
  println(parse_and_double("abc"))
}

/// # Deep Propagation Chain
///
/// Error propagation through 5 nested function layers.
/// The `?` operator should propagate the error all the way up.
///

fn layer1(x: int) -> Result<int, string> {
  if x > 0 { Ok(x) } else { Err("bottom") }
}

fn layer2(x: int) -> Result<int, string> {
  let v = layer1(x)?
  Ok(v + 1)
}

fn layer3(x: int) -> Result<int, string> {
  let v = layer2(x)?
  Ok(v - 1)
}

fn layer4(x: int) -> Result<int, string> {
  let v = layer3(x)?
  Ok(v * 2)
}

fn layer5(x: int) -> Result<int, string> {
  let v = layer4(x)?
  Ok(v / 2)
}

fn main() {
  let a = layer5(1) catch (e) { -1 }
  println(`final: ${string(a)}`)

  let b = layer5(0) catch (e) {
    println(`deep err: ${e}`)
    -1
  }
}

/// # Error Propagate Chain
///
/// Chain multiple `?` operators to propagate errors through layers.
///

fn parse_number(s: string) -> Result<int, string> {
  if s == "42" { Ok(42) }
  else if s == "10" { Ok(10) }
  else { Err("invalid") }
}

fn halve(n: int) -> Result<int, string> {
  if n % 2 != 0 { Err("odd number") } else { Ok(n / 2) }
}


fn parse_and_halve(s: string) -> Result<int, string> {
  let n = parse_number(s)?
  let h = halve(n)?
  Ok(h)
}

fn main() {
  let a = parse_and_halve("42") catch (e) { -1 }

  let b = parse_and_halve("bad") catch (e) {
    println(`caught: ${e}`)
    -1
  }

  println(`result: ${string(a)}`)
}

multiple_question_marks.fg

/// # Multiple Question Marks
///
/// Multiple `?` operators in a single function, each can
/// short-circuit with a different error.
///

fn first(x: int) -> Result<int, string> {
  if x > 0 { Ok(x) } else { Err("first fail") }
}

fn second(x: int) -> Result<int, string> {
  if x < 10 { Ok(x * 2) } else { Err("second fail") }
}

fn third(x: int) -> Result<int, string> {
  if x != 0 { Ok(x + 1) } else { Err("third fail") }
}

fn combined(a: int, b: int, c: int) -> Result<int, string> {
  let x = first(a)?
  let y = second(b)?
  let z = third(c)?
  Ok(x + y + z)
}

fn main() {
  let r1 = combined(1, 2, 1) catch (e) { -1 }
  println(`result: ${string(r1)}`)

  let r2 = combined(1, 20, 1) catch (e) {
    println(`caught: ${e}`)
    -1
  }
}

propagate_in_loop.fg

/// # Propagate In Loop
///
/// The `?` operator inside a while loop exits the function
/// on the first error encountered.
///

fn check(i: int) -> Result<int, string> {
  if i == 3 { Err("fail at 3") } else { Ok(i) }
}

fn sum_until_fail(n: int) -> Result<int, string> {
  mut total = 0
  mut i = 0
  while i < n {
    let v = check(i)?
    total = total + v
    i = i + 1
  }
  Ok(total)
}

fn main() {
  let r = sum_until_fail(10) catch (e) {
    println(`caught: ${e}`)
    -1
  }
}

/// # Result Same Ok Err Type
///
/// Result<string, string> where Ok and Err carry the same type.

  if x { Ok("success") } else { Err("failure") }
