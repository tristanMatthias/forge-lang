if/else — conditional expressions with implicit return

Conditional expressions with if/else and else-if chaining

In Forge, `if`/`else` are expressions, not statements. This means they produce a value and can be used anywhere an expression is expected. For example: `let status = if score > 90 { "A" } else { "B" }`. The last expression in each branch becomes the value of the entire `if` expression.

Conditions do not require parentheses. Write `if x > 0 { ... }` rather than `if (x > 0) { ... }`. The `else if` chain works as expected for multiple conditions: `if x > 0 { "positive" } else if x < 0 { "negative" } else { "zero" }`.

Because `if` is an expression, there is no need for a ternary operator. The expression form is both more readable and more flexible than C-style ternaries. When used as a statement (ignoring the return value), `if` works exactly as you would expect from any other language.

Type checking ensures that both branches of an `if`/`else` return the same type when the result is used as a value. If you write `let x = if cond { 1 } else { "two" }`, the compiler will report a type mismatch.

Syntax

if cond { } else { }
if cond { } else if cond { } else { }

Examples

basic_if.fg

/// # Basic If/Else
///
/// Simple conditional with if/else.
///

fn sign(n: int) -> string {
  if n > 0 {
    "positive"
  } else {
    "zero or negative"
  }
}

fn main() {
  println(sign(5))
  println(sign(-3))
}

if_as_expression.fg

/// # If as Expression
///
/// Using if/else as a value expression in various positions.
///

fn main() {
  let x = if true { "yes" } else { "no" }
  println(x)

  let y = 10 + if true { 20 } else { 0 }
  println(y)

  let label = `result: ${if 100 > 50 { "big" } else { "small" }}`
  println(label)
}

if_complex_conditions.fg

/// # Complex Boolean Conditions
///
/// If with complex &&, || conditions.
///

fn check(a: bool, b: bool) -> string {
  if a && b {
    "both"
  } else if a || b {
    "either"
  } else {
    "neither"
  }
}

fn main() {
  println(check(true, true))
  println(check(true, false))
  println(check(false, false))
}

if_in_match_in_if.fg

/// # If Inside Match Inside If
///
/// Deeply combining if/else with match expressions.
///

fn classify(n: int) -> string {
  if n == 0 {
    "zero"
  } else if n > 0 {
    match n % 2 {
      0 -> if n < 100 { "positive small even" } else { "positive big even" }
      _ -> if n < 100 { "positive small odd" } else { "positive big odd" }
    }
  } else {
    "negative"
  }
}

fn main() {
  println(classify(42))
  println(classify(-5))
  println(classify(0))
}

if_mutation.fg

/// # If with Mutation
///
/// Mutation inside if branches.
///

fn main() {
  mut x = 10
  if true {
    x = x + 5
  }
  println(x)

  mut s = "hello"
  if 1 + 1 == 2 {
    s = s + " world!"
  }
  println(s)
}

stress_nested_if.fg

/// # Deeply Nested If/Else
///
/// If inside if inside if to stress nesting.
///

fn classify(n: int) -> string {
  if n == 0 {
    "zero"
  } else {
    if n > 0 {
      if n < 100 {
        if n % 2 == 0 {
          "small even positive"
        } else {
          "small odd positive"
        }
      } else {
        "big positive"
      }
    } else {
      if n > -100 {
        if n % 2 == 0 {
          "small even negative"
        } else {
          "small odd negative"
        }
      } else {
        if n % 2 == 0 {
          "big even negative"
        } else {
          "big odd negative"
        }
      }
    }
  }
}

fn main() {
  println(classify(42))
  println(classify(-999))
  println(classify(0))
}
