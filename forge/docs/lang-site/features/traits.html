trait/impl — interfaces and polymorphism

Trait declarations with default methods, impl blocks, and associated types

Traits define shared interfaces that types can implement. A trait declares a set of method signatures that implementing types must provide: `trait Printable { fn to_display() -> string }`. Types implement traits with `impl` blocks: `impl Printable for Point { fn to_display() -> string { ... } }`.

Traits enable polymorphism without inheritance. A function that accepts `impl Printable` can work with any type that implements the trait, regardless of the type's other characteristics. This is more flexible than class-based inheritance because a type can implement any number of traits.

Trait bounds on generic type parameters constrain what operations are available: `fn print_all<T: Printable>(items: list<T>)` ensures every item can be displayed. This catches errors at compile time and provides clear documentation about what a function requires.

If you are coming from Go, Forge traits are similar to interfaces. From Rust, they work the same way. From Java or C#, think of them as interfaces with no default methods. From TypeScript, they are like structural interfaces but explicitly declared.

Syntax

trait Name { fn method(self) }
impl Trait for Type { }

Examples

basic_trait.fg

/// # Basic Trait
///
/// Define a trait and implement it for a type.
///

type Point = { x: int, y: int }

trait HasLength {
  fn length(self) -> int
}

impl HasLength for Point {
  fn length(self) -> int {
    self.x + self.y
  }
}

fn main() {
  let p = Point { x: 2, y: 3 }
  println(p.length())
}

///
/// Using traits for type-safe polymorphic

///

trait Animal {
  fn speak(self) -> string
}

type Dog = { name: string }
type Cat = { name: string }

impl Animal for Dog {
  fn speak(self) -> string { "woof!" }
}

impl Animal for Cat {
  fn speak(self) -> string { "meow!" }
}

  println(a.speak())
}

  announce(Dog { name: "Rex" })
  announce(Cat { name: "Whiskers" })
}

/// # Default Trait Methods
///
/// Traits with default method implementations.
///

trait Describable {
  fn describe(self) -> string

  fn describe_loud(self) -> string {
    self.describe().upper()
  }
}

type Circle = { radius: float }
type Square = { side: float }

impl Describable for Circle {
  fn describe(self) -> string { `circle r=${self.radius}` }
}

impl Describable for Square {
  fn describe(self) -> string { `square s=${self.side}` }
}

fn main() {
  let c = Circle { radius: 5.0 }
  let s = Square { side: 3.0 }
  println(c.describe())
  println(c.describe_loud())
  println(s.describe())
  println(s.describe_loud())
}

impl_methods.fg

/// # Impl Methods
///
/// Add methods to a type with `impl`.
///

type Point = { x: int, y: int }

impl Point {
  fn add(self, other: Point) -> Point {
    Point { x: self.x + other.x, y: self.y + other.y }
  }
}

fn main() {
  let a = Point { x: 1, y: 3 }
  let b = Point { x: 2, y: 4 }
  let c = a.add(b)
  println(`(${c.x}, ${c.y})`)
}

operator_overload.fg

/// # Operator Overloading
///
/// Implement Add, Sub, and Display traits for custom types.
///

type Vec2 = { x: float, y: float }

impl Add for Vec2 {
  type Output = Vec2
  fn add(self, rhs: Vec2) -> Vec2 {
    { x: self.x + rhs.x, y: self.y + rhs.y }
  }
}

impl Sub for Vec2 {
  type Output = Vec2
  fn sub(self, rhs: Vec2) -> Vec2 {
    { x: self.x - rhs.x, y: self.y - rhs.y }
  }
}

impl Display for Vec2 {
  fn display(self) -> string {
    `(${self.x}, ${self.y})`
  }
}

impl Eq for Vec2 {
  fn eq(self, other: Vec2) -> bool {
    self.x == other.x && self.y == other.y
  }
}

fn main() {
  let a = Vec2 { x: 1.0, y: 2.0 }
  let b = Vec2 { x: 3.0, y: 4.0 }

  let c = a + b
  println(c.display())

  let d = b - a
  println(d.display())

  println(string(a == a))
  println(string(a == b))
}

/// # Trait Method Override
///
/// Override a default trait method with a custom implementation.
///

trait Scoreable {
  fn base_score(self) -> int

  fn total_score(self) -> int {
    self.base_score() * 2
  }
}

type Player = { score: int }
type Boss = { score: int }

impl Scoreable for Player {
  fn base_score(self) -> int { self.score }
}

impl Scoreable for Boss {
  fn base_score(self) -> int { self.score }

  fn total_score(self) -> int {
    self.base_score() * 10
  }
}

fn main() {
  let p = Player { score: 5 }
  let b = Boss { score: 50 }
  println(string(p.total_score()))
  println(string(b.total_score()))
}

trait_multiple_impl.fg

/// # Multiple Types Implementing Same Trait
///
/// Several types each implementing the same trait.
///

trait Area {
  fn area(self) -> int
}

type Rect = { w: int, h: int }
type Square = { side: int }
type Triangle = { base: int, height: int }

impl Area for Rect {
  fn area(self) -> int { self.w * self.h }
}

impl Area for Square {
  fn area(self) -> int { self.side * self.side }
}

impl Area for Triangle {
  fn area(self) -> int { self.base * self.height / 2 }
}

fn main() {
  let r = Rect { w: 2, h: 5 }
  let s = Square { side: 5 }
  let t = Triangle { base: 3, height: 4 }
  println(string(r.area()))
  println(string(s.area()))
  println(string(t.area()))
}

/// Two different traits implemented on same type, both with default methods
///

trait Printable {
  fn text(self) -> string
  fn loud(self) -> string {
    self.text().upper()
  }
}

trait Measurable {
  fn size(self) -> int
  fn info(self) -> string {
    `length: ${self.size()}`
  }
}

type Word = { content: string }

impl Printable for Word {
  fn text(self) -> string { self.content }
}

impl Measurable for Word {
  fn size(self) -> int { self.content.length }
}

fn main() {
  let w = Word { content: "hello" }
  println(`${w.text()} ${w.loud()}`)
  println(w.info())
}
