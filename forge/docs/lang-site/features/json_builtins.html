JSON Builtins stable

json.parse/stringify — JSON serialization/deserialization

Built-in json.parse(), json.stringify(), and json.parse_list() intrinsics

Forge provides built-in JSON functions: `json.parse(str)` converts a JSON string into a Forge value, and `json.stringify(value)` converts a Forge value into a JSON string. These functions handle the full JSON specification including nested objects, arrays, numbers, strings, booleans, and null.

`json.parse()` returns a dynamic value that can be accessed with field notation and indexing. The parsed structure maps JSON objects to Forge maps, JSON arrays to lists, and JSON primitives to their Forge equivalents (int, float, string, bool, null).

`json.stringify()` serializes any Forge value to its JSON representation. Structs become JSON objects with field names as keys. Lists become JSON arrays. This round-trips cleanly with `json.parse()`, so `json.parse(json.stringify(value))` preserves the structure.

These built-in functions avoid the need for an external JSON library in the vast majority of use cases. They are implemented as intrinsics for maximum performance, with the serialization and deserialization happening in optimized native code.

Syntax

json.parse(str)
json.stringify(val)

Examples

json_all_types.fg

/// # JSON All Types Mixed
///
/// Stringify a struct with string, int, float, bool fields.
///

fn main() {
  let data = { name: "test", count: 42, ratio: 3.14, active: true }
  println(json.stringify(data))
}

json_array_parse.fg

/// # JSON Parse Multiple Fields
///
/// Parse JSON with multiple field types including bool.
///

fn main() {
  let json_str = "{\"name\":\"forge\",\"active\":true,\"version\":3}"
  let data: {name: string, active: bool, version: int} = json.parse(json_str)
  println(data.name)
  println(string(data.active))
  println(string(data.version))
}

/// # Empty JSON Object

fn main() {

json_multiple_nulls.fg

/// # JSON Multiple Null Fields
///
/// Stringify struct where multiple fields are null — all should be omitted.
///

fn main() {
  let a: int? = null
  let b: string? = null
  let c: bool? = null
  let data = { name: "only", a: a, b: b, c: c }
  println(json.stringify(data))
}

json_nested_parse.fg

/// # JSON Nested Parse
///
/// Parse JSON with nested object fields.
///

fn main() {
  let json_str = "{\"name\":\"alice\",\"age\":30}"
  let person: {name: string, age: int} = json.parse(json_str)
  println(person.name)
  println(string(person.age))

  let addr_str = "{\"city\":\"NYC\",\"zip\":10001}"
  let addr: {city: string, zip: int} = json.parse(addr_str)
  println(addr.city)
}

json_parse_list.fg

/// # JSON Parse List of Structs
///
/// Parse a JSON array into a list of typed structs.
///

fn main() {
  let json_str = "[{\"name\":\"alice\",\"age\":30},{\"name\":\"bob\",\"age\":25}]"
  let people: List<{name: string, age: int}> = json.parse(json_str)
  println(string(people.length))
  println(people[0].name)
  println(people[1].name)
}

json_parse_nested.fg

/// # JSON Parse Nested Struct
///
/// json.parse with a nested struct type should correctly parse nested objects.
///

type Address = { city: string }
type Person = { name: string, address: Address }

fn main() {
  let json_str = "{\"name\":\"Alice\",\"address\":{\"city\":\"NYC\"}}"
  let person: Person = json.parse(json_str)
  println(person.name)
  println(person.address.city)
}

json_roundtrip_verify.fg

/// # JSON Roundtrip Verify
///
/// Parse JSON, stringify it back, and verify the result matches.
///

fn main() {
  let original = "{\"name\":\"forge\",\"version\":42}"
  let parsed: {name: string, version: int} = json.parse(original)
  let back = json.stringify(parsed)
  println(string(back == original))
  println(parsed.name)
  println(string(parsed.version))
}

/// # JSON with Special Characters
///
/// Stringify and parse strings containing special characters.

fn main() {

  println(back2.msg)
}
