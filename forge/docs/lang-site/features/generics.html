Generics stable

<T> type parameters with optional trait bounds

Generic type parameters with trait bounds and monomorphization

Generics let you write functions and types that work with any type, while still maintaining full type safety. A generic function is declared with type parameters in angle brackets: `fn identity<T>(x: T) -> T { x }`. The type parameter `T` is replaced with a concrete type at each call site.

Generic types work the same way: `type Wrapper<T> { value: T }` creates a type that can wrap any other type. You can instantiate it as `Wrapper<int> { value: 42 }` or `Wrapper<string> { value: "hi" }`.

Type parameters can be constrained with trait bounds to require certain capabilities. This ensures that generic code can only be called with types that support the operations it needs, catching type errors at compile time rather than runtime.

Forge's generics are similar to those in Rust, TypeScript, and Java. They use monomorphization at compile time, meaning generic code has zero runtime overhead: the compiler generates specialized versions for each concrete type used.

Syntax

fn name<T>(x: T) -> T
fn name<T: Trait>(x: T)

Examples

composition_with_traits.fg

/// # Generic Functions with Trait Bounds
///
/// Type-safe generic programming with trait
/// constraints for compile-time guarantees.
///

trait Answerable {
  fn show(self) -> string
}

type IntVal = { value: int }
type StrVal = { value: string }

impl Answerable for IntVal {
  fn show(self) -> string { string(self.value) }
}

impl Answerable for StrVal {
  fn show(self) -> string { self.value }
}

fn announce<T: Answerable>(item: T) -> string {
  `${item.show()} is the answer`
}

fn main() {
  println(announce(IntVal { value: 42 }))
  println(announce(StrVal { value: "hello" }))
}

generic_bound_multi_type.fg

/// Trait bound with generic: multiple types through same generic fn
///

trait Sound {
  fn sound(self) -> string
}

type Cat = { name: string }
type Dog = { name: string }
type Bird = { name: string }

impl Sound for Cat { fn sound(self) -> string { "meow" } }
impl Sound for Dog { fn sound(self) -> string { "woof" } }
impl Sound for Bird { fn sound(self) -> string { "tweet" } }

fn describe<T: Sound>(animal: T, kind: string) -> string {
  `${kind} says ${animal.sound()}`
}

fn main() {
  println(describe(Cat { name: "c" }, "cat"))
  println(describe(Dog { name: "d" }, "dog"))
  println(describe(Bird { name: "b" }, "bird"))
}

/// Test: Generic functions with list operations
///

  items.length
}

  items.reduce(0, (acc, n) -> acc + n)
}

fn main() {
  let names = ["alice", "bob", "charlie"]
  println(string(count(names)))

  println(names.join(", "))
}

/// # Generic Multiple Parameters
///
/// Function with multiple type parameters.

  println(r2.1)
}

generic_result.fg

/// # Generics with Result
///
/// Generic function returning a Result type.
///

fn safe_first<T>(list: List<T>) -> Result<T, string> {
  if list.length > 0 {
    Ok(list[0])
  } else {
    Err("empty list")
  }
}

fn main() {
  let val = safe_first([100, 200, 300]) catch (e) { 0 }
  println(string(val))

  let empty_val = safe_first([]) catch (e) {
    println(`error: ${e}`)
    0.0
  }
  println(string(empty_val))
}

generic_return_type.fg

/// # Generic Return Type
///
/// Generic used as the return type of a function.
///

fn wrap<T>(val: T) -> T {
  val
}

fn choose<T>(a: T, b: T, pick_first: bool) -> T {
  if pick_first { a } else { b }
}

fn main() {
  let n = wrap(42)
  println(string(n))

  let s = choose("hello", "world", true)
  println(s)
}

trait_bound.fg

/// # Trait Bounds
///
/// Generic functions with trait constraints.
///

trait Describable {
  fn describe(self) -> string
}

type Cat = { name: string }
type Dog = { name: string }

impl Describable for Cat {
  fn describe(self) -> string { `cat: ${self.name}` }
}

impl Describable for Dog {
  fn describe(self) -> string { `dog: ${self.name}` }
}

fn greet<T: Describable>(animal: T) -> string {
  `hello, ${animal.describe()}`
}

fn main() {
  let c = Cat { name: "whiskers" }
  let d = Dog { name: "rex" }
  println(greet(c))
  println(greet(d))
}

/// Trait bound violation: calling generic fn with type that doesn't implement the trait
///

trait Describable {
  fn describe(self) -> string
}

type Dog = { name: string }

fn describe_item<T: Describable>(item: T) -> string {
  item.describe()
}

fn main() {
  let d = Dog { name: "Rex" }
  println(describe_item(d))
}
