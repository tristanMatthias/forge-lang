Spec Test stable

spec/given/then — BDD-style behavior-driven tests

Test framework: spec, given, then, should_fail, where table, skip, todo (requires use @std.test)

Forge includes a built-in BDD-style testing framework with `spec`, `given`, `then`, and `expect` blocks. Tests are written as structured specifications that read like documentation: `spec "math" { given "addition" { then "1 + 1 = 2" { expect(1 + 1 == 2) } } }`.

The `spec` block names the feature being tested. Inside it, `given` blocks describe preconditions or scenarios. `then` blocks describe expected behaviors. `expect(condition)` asserts that a condition is true. This three-level structure organizes tests into a readable hierarchy.

Test output shows the full path of each assertion: `math > addition > 1 + 1 = 2: PASS`. Failed tests show the expected and actual values with source location. The structured output makes it easy to identify exactly which scenario failed and why.

This approach is inspired by RSpec (Ruby), Jest's describe/it (JavaScript), and Kotest (Kotlin). The benefit over flat test functions is that related tests are grouped by topic, and the test names form readable sentences that serve as living documentation.

Syntax

spec "name" { given "setup" { } then "test" { } }

Examples

basic_spec.fg

/// # Basic Spec
///
/// Simple spec with then assertions.
///
/// expect-stderr: ✓ addition works
/// expect-stderr: ✓ multiplication works
/// expect-stderr: 2 passed

use @std.test

fn main() {
    spec "arithmetic" {
        let a = 2
        let b = 3

        then "addition works" { a + b == 5 }
        then "multiplication works" { a * b == 6 }
    }
}

composition_with_closures.fg

/// # Testing Data Transformations
///
/// BDD-style tests verifying collection
/// operations with closures.
///
/// expect-stderr: ✓ double maps correctly
/// expect-stderr: ✓ filter and sum
/// expect-stderr: 2 passed

use @std.test

fn main() {
    spec "list operations" {
        then "double maps correctly" {
            let result = [1, 2, 3].map((x) -> x * 2)
            result.sum() == 12
        }
        then "filter and sum" {
            let result = [1, 2, 3, 4].filter((x) -> x > 2)
            result.sum() == 7
        }
    }
}

deep_nesting.fg

/// # Deep Nesting
///
/// Deeply nested given blocks with setup at each level.
///
/// expect-stderr: ✓ starts at zero
/// expect-stderr: ✓ has one item
/// expect-stderr: ✓ has two items
/// expect-stderr: 3 passed

use @std.test

fn main() {
    spec "counter" {
        given "initial state" {
            let count = 0

            then "starts at zero" { count == 0 }

            given "after first increment" {
                let count = count + 1

                then "has one item" { count == 1 }

                given "after second increment" {
                    let count = count + 1

                    then "has two items" { count == 2 }
                }
            }
        }
    }
}

should_fail.fg

/// # Should Fail
///
/// Tests that are expected to produce falsy results.
///
/// expect-stderr: ✓ negative check
/// expect-stderr: ✓ overflow check
/// expect-stderr: ✗ valid data does not fail
/// expect-stderr: 2 passed
/// expect-stderr: 1 failed
/// expect-exit: 1

use @std.test

fn main() {
    spec "should_fail tests" {
        then "negative check" should_fail {
            -1 > 0
        }

        then "overflow check" should_fail {
            999 < 100
        }

        then "valid data does not fail" should_fail {
            // This succeeds (1 + 1 == 2 is true), so should_fail makes it fail
            1 + 1 == 2
        }
    }
}

spec_with_functions.fg

/// # Spec with Helper Functions
///
/// Spec using helper functions for computation.
///
/// expect-stderr: ✓ factorial of 5 is 120
/// expect-stderr: ✓ factorial of 0 is 1
/// expect-stderr: 2 passed

use @std.test

fn factorial(n: int) -> int {
    if n <= 1 {
        1
    } else {
        n * factorial(n - 1)
    }
}

fn main() {
    spec "factorial" {
        then "factorial of 5 is 120" { factorial(5) == 120 }
        then "factorial of 0 is 1" { factorial(0) == 1 }
    }
}

where_table.fg

/// # Where Table
///
/// Parameterized tests using table literals.
///
/// expect-stderr: ✓ addition
/// expect-stderr: ✓ addition
/// expect-stderr: ✓ addition
/// expect-stderr: 3 passed

use @std.test

fn main() {
    spec "parameterized" {
        then "addition" where table {
            a  | b  | expected
            1  | 2  | 3
            10 | 20 | 30
            0  | 0  | 0
        } {
            a + b == expected
        }
    }
}

where_table_fail.fg

/// # Where Table With Failures
///
/// Parameterized tests with some failing rows.
///
/// expect-stderr: ✓ comparison
/// expect-stderr: ✓ comparison
/// expect-stderr: ✗ comparison
/// expect-stderr: 2 passed
/// expect-stderr: 1 failed
/// expect-exit: 1

use @std.test

fn main() {
    spec "parameterized with failures" {
        then "comparison" where table {
            value | threshold | passes
            10    | 5         | true
            3     | 1         | true
            2     | 5         | true
        } {
            value > threshold == passes
        }
    }
}
