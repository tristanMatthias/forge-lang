Is Keyword stable

value is Pattern — runtime type and pattern checks

Inline pattern check: value is Pattern → bool

The `is` keyword tests whether a value matches a pattern or belongs to a type. It returns a boolean and is used in conditions: `if value is int { ... }` or `if shape is Circle { ... }`. This is the lightweight alternative to a full `match` expression when you only care about one case.

With enums, `is` checks for a specific variant: `if result is Ok { ... }`. Combined with `if let`-style binding, it can extract the associated data: `if result is Ok(value) { ... }`. This handles the common pattern of checking-and-extracting in a single, readable expression.

The `is` keyword also works with nullable types: `if x is null { ... }` checks for null, and `if x is string { ... }` checks that a nullable value is present and of the expected type. This integrates naturally with Forge's null safety system.

Compared to `instanceof` in Java or `typeof` in JavaScript, `is` is a pattern matching operation, not just a type check. It can match literal values, types, enum variants, and complex patterns, making it strictly more powerful.

Syntax

value is Pattern
value is Type

fn main() {
  let a = Ok(5)
  let b = Err("no")

  println(string(a is Ok && b is Err))
  println(string(a is Err || b is Err))
}

composition_with_enums.fg

/// # Type Checks with 'is'
///
/// Using 'is' for enum variant checking
/// combined with conditional logic.
///

enum Shape { circle, square, triangle }

fn describe(s: Shape) -> string {
  if s is .circle {
    "got a circle"
  } else {
    "not a circle"
  }
}

fn main() {
  println(describe(Shape.circle))
  println(describe(Shape.square))
  println(describe(Shape.triangle))
}

enum State { loading, ready, error(msg: string) }

  let s = State.loading

  if s is .loading {
    println("loading")

  println(string(s is .ready))

  let s2 = State.ready
  println(string(s2 is .ready))

  let s3 = State.error(msg: "timeout")
  println(string(s3 is .error))
}

is_in_condition.fg

/// # Is in Complex Conditions
///
/// The `is` keyword used in various conditional contexts.
///

fn process(r: Result<int, string>) -> string {
  if r is Ok {
    "ok value"
  } else {
    "fallback"
  }
}

fn main() {
  println(process(Ok(10)))
  println(process(Err("nope")))

  mut result: Result<int, string> = Err("pending")
  mut tries = 0
  while result is Err {
    tries = tries + 1
    if tries >= 2 {
      result = Ok(42)
    }
  }
  println("done processing")
}

/// # Is Not with Enum Variants
///
/// Use `is not` for negated pattern matching on enum variants.
///

enum Status { pending, active, done }

fn check(s: Status) -> string {
  if s is not .done {
    "not done"
  } else {
    "is done"
  }
}

fn main() {
  println(check(Status.pending))
  println(check(Status.active))
  println(check(Status.done))
}

/// # Is Null Check
///
/// Use `is null` to check nullable values.
///

fn main() {
  let nums = [1, 2, 3]
  let found = nums.find((n) -> n > 10)

  if found is null {
    println("is null")
  } else {
    println("has value")
  }

  let found2 = nums.find((n) -> n > 1)
  if found2 is null {
    println("is null")
  } else {
    println("has value")
  }
}

/// # Is Combined with && and ||
///
/// The `is` keyword used with logical operators in conditions.
///

fn main() {
  let a = Ok(1)
  let b = Ok(2)
  let c = Err("fail")

  if a is Ok && b is Ok {
    println("both ok")
  }

  if a is Err || c is Err {
    println("at least one error")
  }

  if a is Ok && c is Err {
    println("mixed")
  }
}

}

type_variant_check.fg


fn divide(a: int, b: int) -> Result<int, string> {
  if b == 0 { Err("division by zero") } else { Ok(a / b) }
}

fn main() {
  println(string(divide(10, 2) is Ok))
  println(string(divide(10, 0) is Ok))
  println(string(divide(10, 0) is Err))
}
