# Forge — Package Component System

## Overview

A component block has four concepts:

| Concept | User writes | Package defines |
|---|---|---|
| **Config** | `key value` | `config { key: type = default }` |
| **Events** | `on event(args) { }` | `event name(args)` — declares the slot |
| **Functions** | function calls or syntax sugar | `fn name(args) { }` with optional `@syntax` |
| **Lifecycle** | — | `on startup { }`, `on main_end { }` |

Config is `identifier simple_literal` (bool, number, string, duration). Events are declared by the package with `event`, filled by the user with `on`. Functions are called normally or via `@syntax` sugar. Lifecycle hooks are package-only.

---

## Defining a Component

```forge
component server(port: int) {
  // ── Config: typed, validated, with defaults ──
  config {
    cors: bool = false
    logging: bool = false
    rate_limit: int = 0
  }

  // ── Events: declare slots the user can hook into ──
  event error(err: string, req: Request)

  // ── Setup: runs before user's block body ──
  let id = forge_http_create(port)
  if config.cors { forge_http_enable_cors(id) }
  if config.logging { forge_http_enable_logging(id) }
  if config.rate_limit > 0 { forge_http_set_rate_limit(id, config.rate_limit) }
  forge_http_set_error_handler(id, error)   // pass event to native lib

  // ── Functions: available inside the block ──
  // Can be called as normal functions: route("GET", "/health", handler)
  // Or via syntax sugar: GET /health -> handler

  @syntax("{method} {path} -> {handler}")
  fn route(method: string, path: string, handler: fn(Request) -> any) {
    forge_http_add_route(id, method, path, req -> json.stringify(handler(req)))
  }

  @syntax("{service} at {path}")
  fn mount(service: Service, path: string) {
    route("GET", path, req -> service.list())
    route("GET", `${path}/:id`, req -> service.get(req.params.id))
    route("POST", path, req -> service.create(req.body))
    route("PUT", `${path}/:id`, req -> service.update(req.params.id, req.body))
    route("DELETE", `${path}/:id`, req -> service.delete(req.params.id))
  }

  // ── User's block body inserted here automatically ──

  // ── Lifecycle ──
  on main_end { forge_http_serve(id) }
}
```

## Using a Keyword

Both styles work — sugar or function calls:

```forge
// Sugar
server :8080 {
  cors true
  GET /health -> { status: "ok" }
  GET /users/:id -> (req) { Users.get(req.params.id) }
  mount TodoService at /todos
  on error(err) { log(err) }
}

// Function calls
server :8080 {
  cors true
  route("GET", "/health", (req) -> { status: "ok" })
  route("GET", "/users/:id", (req) -> Users.get(req.params.id))
  mount(TodoService, "/todos")
  on error(err) { log(err) }
}

// Mixed
server :8080 {
  cors true
  GET /health -> { status: "ok" }
  mount(TodoService, "/todos")
}
```

## More Examples

### Model

```forge
component model(name: string) {
  config {
    table_name: string = name.lower() + "s"
  }

  // Events — used by service component to wrap CRUD with hooks
  event before_create(record)
  event after_create(record)
  event before_update(record, changes)
  event after_update(record)
  event before_delete(id: int)
  event after_delete(id: int)

  @syntax("{name}: {type}")
  fn field(name: string, type: Type, annotations: List<Annotation>) {
    add_column(config.table_name, name, type, annotations)
  }

  on startup {
    forge_model_init(config("database.path", "./data/forge.db"))
    forge_model_exec(generate_create_table_sql(config.table_name, fields))
  }

  fn ${name}.create(data) -> ${name} {
    before_create(data)          // calls event (no-op if user didn't define it)
    let result = json.parse(forge_model_insert(config.table_name, json.stringify(data)))
    after_create(result)
    result
  }

  fn ${name}.get(id: int) -> ${name}? {
    let raw = forge_model_get_by_id(config.table_name, id)
    if raw == "" { null } else { json.parse(raw) }
  }

  fn ${name}.list() -> List<${name}> {
    json.parse(forge_model_list(config.table_name, "{}"))
  }

  fn ${name}.delete(id: int) -> bool {
    before_delete(id)
    let result = forge_model_delete(config.table_name, id) > 0
    after_delete(id)
    result
  }
}
```

User writes:

```forge
model User {
  id: int @primary @auto_increment
  name: string
  email: string @unique
  active: bool @default(true)

  on before_create(user) {
    assert user.name.length > 0, "name required"
  }

  on after_create(user) {
    println(`created: ${user.name}`)
  }
}
```

Each `name: type @annotation` line matches the `@syntax("{name}: {type}")` pattern and calls `field(...)`.

### Agent

```forge
component agent(name: string) {
  config {
    model: string = "claude-sonnet"
    temperature: float = 0.7
    max_tokens: int = 4096
  }

  fn tool(name: string, params: List<Param>, return_type: Type, body: Block) {
    forge_ai_register_tool(id, name, params, body)
  }

  fn system_prompt(body: Block) {
    forge_ai_set_system_prompt(id, body.text)
  }

  let id = forge_ai_create(name, config.model, config.temperature)

  on message(input) -> stream {
    forge_ai_respond(id, input)
  }
}
```

### Queue

```forge
component queue(name: string) {
  config {
    retries: int = 3
    buffer_size: int = 1000
    backoff: string = "linear"
  }

  // Events — user hooks into these with `on`
  event message(msg: string)
  event error(err: string, msg: string)

  let id = forge_queue_create(name, config.buffer_size)

  fn send(data: json) {
    forge_queue_send(id, json.stringify(data))
  }

  fn receive() -> json? {
    let raw = forge_queue_receive(id)
    if raw == "" { null } else { json.parse(raw) }
  }

  fn depth() -> int {
    forge_queue_depth(id)
  }

  // Pass event handlers to native lib — it calls them when messages arrive
  on startup {
    forge_queue_start_worker(id, message, error)
  }
}
```

## @syntax Rules

1. Patterns are `literal {param} literal {param}` — placeholders between fixed tokens
2. No optional segments, no nesting, no wildcards, no regex
3. Patterns must be unambiguous within a component block — two patterns can't match the same input
4. Function call syntax always works as fallback — sugar is optional
5. `{param}` names must match the function's parameter names

---

## Parser Disambiguation

Inside a component block, the compiler categorizes each line:

| Pattern | Category |
|---|---|
| `identifier simple_literal` | Config (`cors true`, `retries 3`, `timeout 30s`) |
| `on identifier(args) { }` | Event handler (user fills a declared event slot) |
| `identifier(args)` | Function call |
| matches an `@syntax` pattern | Sugar — desugars to function call |
| nothing matches | Compile error |

Simple literals: `true`, `false`, integers, floats, strings, durations. Nothing compound.

Events declared with `event name(args)` in the package are callable as functions within the component expansion. If the user doesn't define an `on` handler for an event, calls to it are no-ops.

## Package File Structure

```
my-package/
├── package.toml
├── src/
│   └── package.fg       # extern fn, types, components, helpers
└── lib/
    └── <platform>/
        └── lib*.a         # Native library (any compiled language)
```

Library packages (no components): just `extern fn` + exported Forge wrapper functions. No `package.toml` component section needed.

Component packages: add `component` blocks with `config`, `fn`, `@syntax`, `event`, and `on` lifecycle hooks.
